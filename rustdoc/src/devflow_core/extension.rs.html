12/// The action an extension wishes to execute for a given command.
13#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
14pub struct ExecutionAction {
15    /// The executable program (e.g., "cargo", "npm").
16    pub program: String,
17    /// The arguments to pass to the program.
18    pub args: Vec<String>,
19    /// Optional environment variables to set for the execution.
20    #[serde(default)]
21    pub env: HashMap<String, String>,
22}
23
24/// A contract for all extensions connecting to Devflow.
25pub trait Extension: std::fmt::Debug {
26    /// Unique name of the extension.
27    fn name(&self) -> &str;
28    /// The set of command capabilities provided by this extension.
29    fn capabilities(&self) -> HashSet<String>;
30    /// Maps a command reference to an executable action.
31    fn build_action(&self, cmd: &CommandRef) -> Result<Option<ExecutionAction>>;
32
33    /// Whether this extension is considered "trusted" to run on the host during negotiation.
34    fn is_trusted(&self) -> bool {
35        false
36    }
37
38    /// Returns the host-to-container volume mappings required by this extension.
39    /// Expected format: `host_relative_dir:container_absolute_dir`
40    /// Example: `rust/cargo:/usr/local/cargo`
41    fn cache_mounts(&self) -> Vec<String> {
42        Vec::new()
43    }
44
45    /// Returns the environment variables required by this extension for execution.
46    fn env_vars(&self) -> HashMap<String, String> {
47        HashMap::new()
48    }
49
50    /// Returns a list of files or globs that constitute the execution fingerprint identity.
51    /// Example: `["rust-toolchain.toml", "Cargo.lock"]`
52    fn fingerprint_inputs(&self) -> Vec<String> {
53        Vec::new()
54    }
55}

57/// A registry containing all discovered Devflow extensions.
58#[derive(Debug, Default)]
59pub struct ExtensionRegistry {
60    extensions: HashMap<String, Box<dyn Extension>>,
61}
62
63impl ExtensionRegistry {
64    /// Discovers extensions based on the provided configuration.
65    ///
66    /// If no extensions are explicitly configured, it attempts to load
67    /// builtin extensions based on the project stack.
68    pub fn discover(config: &DevflowConfig) -> Result<Self> {
69        debug!(
70            "discovering extensions for project: {}",
71            config.project.name
72        );
73        let registry = Self::default();
74        // Discovery logic will be delegated to the CLI / wiring phase,
75        // but for backwards capability during refactor, we accept the config
76        // even if we leave the registry empty.
77        Ok(registry)
78    }
79
80    /// Registers a new extension into the registry.
81    pub fn register(&mut self, extension: Box<dyn Extension>) {
82        self.extensions
83            .insert(extension.name().to_string(), extension);
84    }
85
86    /// Retrieves an extension by name.
87    pub fn get(&self, name: &str) -> Option<&dyn Extension> {
88        self.extensions.get(name).map(|boxed| boxed.as_ref())
89    }
90
91    /// Verifies if any registered extension can handle the given command.
92    ///
93    /// # Errors
94    /// Returns an error if no registered extension exposes the required capability.
95    #[instrument(skip(self))]
96    pub fn ensure_can_run(&self, cmd: &CommandRef) -> Result<()> {
97        debug!("checking capability support for: {}", cmd.canonical());
98        if self.extensions.is_empty() {
99            // Command planning still works without explicit extensions for early bootstrap.
100            return Ok(());
101        }
102
103        let selector_key = cmd
104            .selector
105            .as_ref()
106            .map(|selector| format!("{}:{}", cmd.primary.as_str(), selector));
107        let primary_key = cmd.primary.as_str().to_string();
108
109        let supported = self.extensions.values().any(|ext| {
110            ext.capabilities().contains(&primary_key)
111                || selector_key
112                    .as_ref()
113                    .map(|s| ext.capabilities().contains(s))
114                    .unwrap_or(false)
115        });
116
117        if supported {
118            return Ok(());
119        }
120
121        bail!(
122            "no extension exposes capability '{}'",
123            selector_key.unwrap_or(primary_key)
124        )
125    }
126
127    /// Validates that all commands defined in the project targets are supported by at least one extension.
128    pub fn validate_target_support(&self, cfg: &DevflowConfig) -> Result<()> {
129        if self.extensions.is_empty() {
130            return Ok(());
131        }
132
133        for (profile, commands) in &cfg.targets.profiles {
134            for raw in commands {
135                let cmd = CommandRef::from_str(raw)?;
136                self.ensure_can_run(&cmd).map_err(|e| {
137                    anyhow::anyhow!(
138                        "unsupported command '{}' in targets profile '{}': {}",
139                        raw,
140                        profile,
141                        e
142                    )
143                })?;
144            }
145        }
146
147        Ok(())
148    }
149
150    /// Builds the execution arguments for a command against a specific extension.
151    pub fn build_action(&self, name: &str, cmd: &CommandRef) -> Result<Option<ExecutionAction>> {
152        if let Some(ext) = self.extensions.get(name) {
153            let mut action = match ext.build_action(cmd)? {
154                Some(a) => a,
155                None => return Ok(None),
156            };
157            // Merge extension global envs with action-specific envs
158            let mut merged_env = ext.env_vars();
159            merged_env.extend(action.env);
160            action.env = merged_env;
161            Ok(Some(action))
162        } else {
163            Ok(None)
164        }
165    }
166
167    /// Aggregates all cache mounts requested by the active extensions.
168    /// Used by the container executor to map generic host directories.
169    pub fn all_cache_mounts(&self) -> Vec<String> {
170        let mut mounts = HashSet::new();
171        for ext in self.extensions.values() {
172            for mount in ext.cache_mounts() {
173                mounts.insert(mount);
174            }
175        }
176        let mut sorted: Vec<String> = mounts.into_iter().collect();
177        sorted.sort();
178        sorted
179    }
180}

380    #[test]
381    fn build_action_merges_env_vars_with_action_override() {
382        let mut registry = ExtensionRegistry::default();
383
384        let mut ext_envs = HashMap::new();
385        ext_envs.insert("CARGO_HOME".to_string(), "/default/cargo".to_string());
386        ext_envs.insert("CI".to_string(), "true".to_string());
387
388        let mut action_envs = HashMap::new();
389        action_envs.insert("CARGO_HOME".to_string(), "/override/cargo".to_string());
390        action_envs.insert("EXTRA".to_string(), "value".to_string());
391
392        registry.register(Box::new(ConfigurableMockExtension {
393            ext_name: "rust".to_string(),
394            capabilities: HashSet::new(),
395            action: Some(ExecutionAction {
396                program: "cargo".to_string(),
397                args: vec!["build".to_string()],
398                env: action_envs,
399            }),
400            trusted: true,
401            mounts: Vec::new(),
402            envs: ext_envs,
403        }));
404
405        let cmd = CommandRef {
406            primary: PrimaryCommand::Build,
407            selector: None,
408        };
409
410        let action = registry.build_action("rust", &cmd).unwrap().unwrap();
411        // Action-level env overrides extension-level
412        assert_eq!(action.env.get("CARGO_HOME").unwrap(), "/override/cargo");
413        // Extension-level env is preserved when not overridden
414        assert_eq!(action.env.get("CI").unwrap(), "true");
415        // Action-specific env is preserved
416        assert_eq!(action.env.get("EXTRA").unwrap(), "value");
417    }
