11/// The root configuration structure for a Devflow project.
12///
13/// This structure is typically deserialized from a `devflow.toml` file.
14#[derive(Debug, Deserialize)]
15#[serde(deny_unknown_fields)]
16pub struct DevflowConfig {
17    /// Basic project metadata.
18    pub project: ProjectConfig,
19    /// Runtime settings (e.g., local, CI).
20    #[serde(default)]
21    pub runtime: RuntimeConfig,
22    /// Custom target profiles (e.g., `pr`, `main`, `release`).
23    #[serde(default)]
24    pub targets: TargetsConfig,
25    /// Optional extension configurations.
26    pub extensions: Option<HashMap<String, ExtensionConfig>>,
27    /// Container configuration for execution proxies.
28    #[serde(default)]
29    pub container: Option<ContainerConfig>,
30    /// Cache configuration for build artifact management.
31    #[serde(default)]
32    pub cache: Option<CacheConfig>,
33    /// Path to the directory containing this config file, used to anchor relative paths.
34    #[serde(skip)]
35    pub source_dir: Option<PathBuf>,
36}
37
38impl DevflowConfig {
39    /// Loads a `DevflowConfig` from a TOML file at the given path.
40    ///
41    /// # Errors
42    /// Returns an error if the file cannot be read, the TOML is invalid,
43    /// or the configuration fails validation.
44    pub fn load_from_file(path: &str) -> Result<Self> {
45        let text = std::fs::read_to_string(path)
46            .with_context(|| format!("failed to read config file: {path}"))?;
47        let mut cfg = toml::from_str::<Self>(&text)
48            .with_context(|| format!("failed to parse TOML config: {path}"))?;
49
50        cfg.source_dir = Some(
51            PathBuf::from(path)
52                .parent()
53                .unwrap_or(std::path::Path::new(""))
54                .to_path_buf(),
55        );
56        cfg.validate()?;
57        Ok(cfg)
58    }
59
60    /// Validates the configuration for logical consistency.
61    fn validate(&self) -> Result<()> {
62        // Devflow Core is stack-agnostic. We allow any stack name here, as long as
63        // an extension (builtin or subprocess) registers to handle it during runtime execution.
64
65        for (profile, commands) in &self.targets.profiles {
66            for raw in commands {
67                CommandRef::from_str(raw).map_err(|e| {
68                    anyhow!(
69                        "invalid command '{}' in targets profile '{}': {}",
70                        raw,
71                        profile,
72                        e
73                    )
74                })?;
75            }
76        }
77
78        Ok(())
79    }
80}

111/// Configuration for containerized execution environments.
112#[derive(Debug, Clone, Deserialize, Default)]
113#[serde(deny_unknown_fields)]

116    /// If not provided, a default base image may be used.

118    /// The container engine to use (e.g., "docker", "podman").

124}

146/// Configuration for an individual extension.
147#[derive(Debug, Clone, Deserialize)]
148#[serde(deny_unknown_fields)]
149pub struct ExtensionConfig {
150    /// Where the extension is sourced from (builtin or path).
151    pub source: ExtensionSource,
152    /// Optional path for path-sourced extensions.
153    pub path: Option<PathBuf>,
154    /// Optional version string.
155    pub version: Option<String>,
156    /// The API version the extension expects.
157    pub api_version: Option<u32>,
158    /// List of capabilities exposed by the extension.
159    #[serde(default)]
160    pub capabilities: Vec<String>,
161    /// Whether this extension is required for project operations.
162    #[serde(default)]
163    pub required: bool,
164    /// Whether this extension is trusted to run on the host during negotiation.
165    #[serde(default)]
166    pub trusted: bool,
167}
168
169/// Source types for extensions.
170#[derive(Debug, Clone, Deserialize)]
171#[serde(rename_all = "lowercase")]
172pub enum ExtensionSource {
173    /// A builtin extension bundled with the Devflow binary.
174    Builtin,
175    /// An extension loaded from a local directory.
176    Path,
177}
