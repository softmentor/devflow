8/// Runs the `init` command to bootstrap a new Devflow project.
9#[instrument(skip(cli))]
10pub fn run(cli: &Cli, template_selector: Option<&str>) -> Result<()> {
11    let config_path = Path::new(&cli.config);
12    let parent = config_path.parent().unwrap_or_else(|| Path::new(""));
13    let target_dir = if parent.as_os_str().is_empty() {
14        std::env::current_dir()?
15    } else {
16        parent.to_path_buf()
17    };
18
19    let project_name = target_dir
20        .file_name()
21        .map(|n| n.to_string_lossy().to_string())
22        .unwrap_or_else(|| "devflow-project".to_string());
23
24    let template = match template_selector {
25        Some(value) => InitTemplate::from_str(value)?,
26        None => detect_template(&target_dir)?,
27    };
28
29    let config_content = template.render_config(&project_name);
30    write_if_absent(&cli.config, &config_content, cli.force)
31        .with_context(|| format!("failed to write '{}'", cli.config))?;
32
33    if let Some(dockerfile_content) = template.render_dockerfile() {
34        let dockerfile_path = target_dir.join("Dockerfile.devflow");
35        let df_str = dockerfile_path.to_str().unwrap();
36        // Ignore failures if the Dockerfile already exists, this is non-critical scaffolding
37        let _ = write_if_absent(df_str, dockerfile_content, cli.force);
38    }
39
40    let cfg = devflow_core::DevflowConfig::load_from_file(&cli.config)?;
41    let workflow = devflow_gh::render_workflow(&cfg)?;
42
43    if cli.stdout {
44        println!("{workflow}");
45    } else {
46        write_if_absent(&cli.ci_output, &workflow, cli.force)
47            .with_context(|| format!("failed to write '{}'", cli.ci_output))?;
48    }
49
50    info!(
51        "init complete: template={}, config={}, ci={}",
52        template.as_str(),
53        cli.config,
54        cli.ci_output
55    );
56    println!("next: run 'dwf check:pr'");
57
58    Ok(())
59}

61/// Supported project templates for initialization.
62#[derive(Debug, Clone, Copy)]
63enum InitTemplate {
64    /// Standard Rust project.
65    Rust,
66    /// Node.js project.
67    Node,
68    /// TypeScript project with common defaults.
69    Tsc,
70    /// Kotlin project (custom stack example).
71    Kotlin,
72}
73
74impl InitTemplate {
75    fn from_str(value: &str) -> Result<Self> {
76        match value {
77            "rust" => Ok(Self::Rust),
78            "node" => Ok(Self::Node),
79            "tsc" | "typescript" => Ok(Self::Tsc),
80            "kotlin" => Ok(Self::Kotlin),
81            other => Err(anyhow!(
82                "unknown init template '{}' (supported: rust,node,tsc,kotlin)",
83                other
84            )),
85        }
86    }
87
88    fn as_str(self) -> &'static str {
89        match self {
90            Self::Rust => "rust",
91            Self::Node => "node",
92            Self::Tsc => "tsc",
93            Self::Kotlin => "kotlin",
94        }
95    }
96
97    fn render_config(self, project_name: &str) -> String {
98        let template = match self {
99            Self::Rust => include_str!("../resources/rust.toml"),
100            Self::Node => include_str!("../resources/node.toml"),
101            Self::Tsc => include_str!("../resources/tsc.toml"),
102            Self::Kotlin => include_str!("../resources/kotlin.toml"),
103        };
104
105        template
106            .replace("my-rust-project", project_name)
107            .replace("my-node-project", project_name)
108            .replace("my-typescript-project", project_name)
109            .replace("my-kotlin-project", project_name)
110    }
111
112    fn render_dockerfile(self) -> Option<&'static str> {
113        match self {
114            Self::Rust => Some(include_str!("../resources/Dockerfile.rust")),
115            Self::Node | Self::Tsc => Some(include_str!("../resources/Dockerfile.node")),
116            Self::Kotlin => None,
117        }
118    }
119}

121fn detect_template(base_path: &Path) -> Result<InitTemplate> {
122    use devflow_core::constants::{MANIFEST_NODE, MANIFEST_RUST, MANIFEST_TSC};
123
124    if base_path.join(MANIFEST_RUST).exists() {
125        return Ok(InitTemplate::Rust);
126    }
127
128    if base_path.join(MANIFEST_TSC).exists() {
129        return Ok(InitTemplate::Tsc);
130    }
131
132    if base_path.join(MANIFEST_NODE).exists() {
133        return Ok(InitTemplate::Node);
134    }
135
136    Err(anyhow!(
137        "unable to auto-detect template. Run: dwf init <rust|node|tsc|kotlin>"
138    ))
139}

141fn write_if_absent(path: &str, content: &str, force: bool) -> Result<()> {
142    let output = Path::new(path);
143
144    if output.exists() && !force {
145        return Err(anyhow!(
146            "'{}' already exists. Re-run with --force to overwrite",
147            path
148        ));
149    }
150
151    if let Some(parent) = output.parent() {
152        fs::create_dir_all(parent)
153            .with_context(|| format!("failed to create directory '{}'", parent.display()))?;
154    }
155
156    fs::write(output, content)
157        .with_context(|| format!("failed to write file '{}'", output.display()))
158}

187        let base = dir.path();

201    #[test]
202    fn unit_test_write_if_absent() {
203        let dir = tempdir().unwrap();
204        let file_path = dir.path().join("test.txt").to_str().unwrap().to_string();
205
206        // Successful write
207        assert!(write_if_absent(&file_path, "hello", false).is_ok());
208        assert_eq!(fs::read_to_string(&file_path).unwrap(), "hello");
209
210        // Fail to overwrite existing file without force
211        assert!(write_if_absent(&file_path, "world", false).is_err());
212        assert_eq!(fs::read_to_string(&file_path).unwrap(), "hello");
213
214        // Successful overwrite with force
215        assert!(write_if_absent(&file_path, "world", true).is_ok());
216        assert_eq!(fs::read_to_string(&file_path).unwrap(), "world");
217    }
