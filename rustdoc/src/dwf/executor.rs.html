3//! This module handles the dispatch of Devflow commands to their respective

18/// Default image used for containerized execution if none specified.
19const DEFAULT_CI_IMAGE: &str = "ghcr.io/softmentor/devflow-ci:latest";
20/// Default host directory for the Devflow cache.
21const DEFAULT_CACHE_ROOT: &str = ".cache/devflow";
22/// The internal container path where the project is mounted.
23const CONTAINER_WORKSPACE: &str = "/workspace";
24/// The internal container path where the host `dwf` binary is mapped.
25const CONTAINER_DWF_BIN: &str = "/usr/local/bin/dwf";

27/// Runs a Devflow command by dispatching it to applicable stacks.
28#[instrument(skip(cfg, registry), fields(command = %command))]
29pub fn run(cfg: &DevflowConfig, registry: &ExtensionRegistry, command: &CommandRef) -> Result<()> {
30    let mut attempted = false;
31
32    let mut requested_stacks = Vec::new();
33    for stack in &cfg.project.stack {
34        if stack_is_applicable(cfg, stack) {
35            requested_stacks.push(stack.clone());
36        } else {
37            info!(target: "devflow", "skip {}: manifest not found", stack);
38        }
39    }
40
41    if let Some(extensions) = &cfg.extensions {
42        for ext_name in extensions.keys() {
43            if !requested_stacks.contains(ext_name) {
44                // Explicitly declared subprocess extensions assume implicit applicability
45                requested_stacks.push(ext_name.clone());
46            }
47        }
48    }
49
50    for stack in &requested_stacks {
51        let effective = with_default_selector(command);
52
53        let is_already_in_container = std::env::var("IS_CONTAINER")
54            .map(|v| v == "true")
55            .unwrap_or(false);
56
57        let action_result = if cfg.runtime.profile == RuntimeProfile::Container
58            && !is_already_in_container
59        {
60            // If we are in container mode, check if the extension is trusted to run on host
61            if let Some(ext) = registry.get(stack) {
62                if ext.is_trusted() {
63                    map_command(stack, &effective, registry)
64                } else {
65                    // UNTRUSTED negotiation: we must run build_action inside a container
66                    // For now, we bail with a helpful message until the full Pre-Flight container is wired
67                    bail!("untrusted extension '{}' cannot negotiate on host in container mode. Move to trusted = true or wait for Pre-Flight jail support.", stack);
68                }
69            } else {
70                map_command(stack, &effective, registry)
71            }
72        } else {
73            map_command(stack, &effective, registry)
74        };
75
76        let Some(action) = action_result? else {
77            info!(target: "devflow",
78                "skip {}: unsupported command {}",
79                stack,
80                effective.canonical()
81            );
82            continue;
83        };
84
85        attempted = true;
86
87        // When IS_CONTAINER=true (e.g., inside GHA native container: job),
88        // skip the docker-run proxy even if profile is "container".
89        // This enables GHA native container jobs to run dwf commands directly.
90        let is_already_in_container = std::env::var("IS_CONTAINER")
91            .map(|v| v == "true")
92            .unwrap_or(false);
93
94        let use_container_proxy =
95            cfg.runtime.profile == RuntimeProfile::Container && !is_already_in_container;
96        let final_action = if use_container_proxy {
97            build_container_proxy(cfg, registry, &action)?
98        } else {
99            sanitize_host_env(action)
100        };
101
102        info!(target: "devflow", "run {} on {}", effective, stack);
103        run_action(&final_action)
104            .with_context(|| format!("{} failed for {}", effective.canonical(), stack))?;
105    }
106
107    if !attempted {
108        bail!(
109            "command '{}' did not match any runnable stack",
110            command.canonical()
111        );
112    }
113
114    Ok(())
115}

117/// Removes container-bound env values when running directly on host.
118///
119/// Extensions may return envs like `/workspace/...` or `/root/...` for container parity.
120/// In host profile these paths are often invalid or read-only and can break local execution.
121fn sanitize_host_env(mut action: ExecutionAction) -> ExecutionAction {
122    action.env.retain(|key, value| {
123        if matches!(
124            key.as_str(),
125            "CARGO_HOME"
126                | "CARGO_TARGET_DIR"
127                | "SCCACHE_DIR"
128                | "RUSTC_WRAPPER"
129                | "NPM_CONFIG_CACHE"
130        ) {
131            return false;
132        }
133
134        !(value == "/workspace"
135            || value.starts_with("/workspace/")
136            || value == "/root"
137            || value.starts_with("/root/"))
138    });
139    action
140}

142/// Normalizes a command by applying default selectors if missing.
143fn with_default_selector(command: &CommandRef) -> CommandRef {
144    if command.selector.is_some() {
145        return command.clone();
146    }
147
148    CommandRef {
149        primary: command.primary,
150        selector: Some(command.primary.default_selector().to_string()),
151    }
152}

157    devflow_core::project::stack_is_applicable(base, stack)

160/// Maps a logical Devflow command to a concrete execution action for a given stack.
161fn map_command(
162    stack: &str,
163    cmd: &CommandRef,
164    registry: &ExtensionRegistry,
165) -> Result<Option<ExecutionAction>> {
166    match stack {
167        "custom" => Ok(map_custom(cmd)),
168        _ => registry.build_action(stack, cmd),
169    }
170}

172/// Fallback logic for projects using `Makefile` or `justfile` without a specific Devflow extension.
173fn map_custom(cmd: &CommandRef) -> Option<ExecutionAction> {
174    let target = cmd.canonical().replace(':', "-");
175
176    if Path::new("justfile").exists() && command_exists("just") {
177        return Some(ExecutionAction {
178            program: "just".to_string(),
179            args: vec![target],
180            env: std::collections::HashMap::new(),
181        });
182    }
183    if Path::new("Makefile").exists() {
184        return Some(ExecutionAction {
185            program: "make".to_string(),
186            args: vec![target],
187            env: std::collections::HashMap::new(),
188        });
189    }
190
191    match (cmd.primary, cmd.selector.as_deref().unwrap_or("")) {
192        (PrimaryCommand::Setup, "doctor") => Some(ExecutionAction {
193            program: "echo".to_string(),
194            args: vec!["custom stack requires justfile or Makefile targets".to_string()],
195            env: std::collections::HashMap::new(),
196        }),
197        _ => None,
198    }
199}

201/// Executes a process on the host system.
202fn run_action(action: &ExecutionAction) -> Result<()> {
203    let status = Command::new(&action.program)
204        .args(&action.args)
205        .envs(action.env.iter())
206        .status()
207        .with_context(|| {
208            format!(
209                "failed to start command '{} {}'",
210                action.program,
211                action.args.join(" ")
212            )
213        })?;
214
215    if !status.success() {
216        bail!(
217            "command failed with status {}: {} {}",
218            status,
219            action.program,
220            action.args.join(" ")
221        );
222    }
223
224    Ok(())
225}

227/// Transforms a host execution action into a containerized proxy action.
228///
229/// This involves:
230/// 1. Detecting an available container engine (Docker/Podman).
231/// 2. Resolving the appropriate container image.
232/// 3. Injecting the host `dwf` binary into the container to ensure version parity.
233/// 4. Mounting the workspace and any extension-defined cache volumes.
234fn build_container_proxy(
235    cfg: &DevflowConfig,
236    registry: &ExtensionRegistry,
237    action: &ExecutionAction,
238) -> Result<ExecutionAction> {
239    let container_config = cfg.container.as_ref();
240    let engine_cfg = container_config.map(|c| c.engine).unwrap_or_default();
241
242    let engine_cmd = resolve_engine(engine_cfg)?;
243
244    let image = container_config
245        .and_then(|c| c.image.clone())
246        .unwrap_or_else(|| DEFAULT_CI_IMAGE.to_string());
247
248    let dwf_cache_root = std::env::var("DWF_CACHE_ROOT")
249        .ok()
250        .or_else(|| cfg.cache.as_ref().and_then(|c| c.root.clone()))
251        .unwrap_or_else(|| DEFAULT_CACHE_ROOT.to_string());
252
253    let cwd = std::env::current_dir()?;
254    let cwd_str = cwd.to_string_lossy();
255
256    // Version parity safety: we map the host's actively executing `dwf` binary
257    // into the container so that even if the container image is old, it always
258    // uses the exact same Devflow logic as the invoker.
259    let host_dwf_path = std::env::current_exe()?;
260    let host_dwf_str = host_dwf_path.to_string_lossy();
261
262    let mut args = vec![
263        "run".to_string(),
264        "--rm".to_string(),
265        "-v".to_string(),
266        format!("{}:{}", cwd_str, CONTAINER_WORKSPACE),
267        "-v".to_string(),
268        format!("{}:{}:ro", host_dwf_str, CONTAINER_DWF_BIN),
269        "-w".to_string(),
270        CONTAINER_WORKSPACE.to_string(),
271    ];
272
273    // Cache redirection: extensions define relative paths (e.g. ".cargo") which
274    // we anchor to the unified `DWF_CACHE_ROOT` on the host.
275    let abs_cache_root = resolve_cache_root(cfg, &dwf_cache_root);
276    let mounts = registry.all_cache_mounts();
277
278    for mount in mounts {
279        if let Some((host_rel, container_abs)) = parse_mount(&mount) {
280            let host_abs = abs_cache_root.join(host_rel);
281
282            if let Err(e) = std::fs::create_dir_all(&host_abs) {
283                warn!(
284                    "failed to create cache directory {}: {}",
285                    host_abs.display(),
286                    e
287                );
288            }
289
290            args.push("-v".to_string());
291            args.push(format!("{}:{}", host_abs.display(), container_abs));
292        } else {
293            warn!("invalid cache mount format from extension: {}", mount);
294        }
295    }
296
297    for (key, value) in &action.env {
298        args.push("-e".to_string());
299        args.push(format!("{}={}", key, value));
300    }
301
302    args.push(image);
303    args.push(action.program.clone());
304    args.extend(action.args.clone());
305
306    Ok(ExecutionAction {
307        program: engine_cmd,
308        args,
309        env: action.env.clone(),
310    })
311}

313fn resolve_engine(engine_cfg: ContainerEngine) -> Result<String> {
314    let cmd = match engine_cfg {
315        ContainerEngine::Docker => "docker",
316        ContainerEngine::Podman => "podman",
317        ContainerEngine::Auto => {
318            if is_engine_healthy("podman") {
319                "podman"
320            } else if is_engine_healthy("docker") {
321                "docker"
322            } else if command_exists("podman") {
323                "podman"
324            } else if command_exists("docker") {
325                "docker"
326            } else {
327                bail!("no container engine (docker or podman) found on PATH");
328            }
329        }
330    };
331
332    if !command_exists(cmd) {
333        bail!("required container engine '{cmd}' is not installed or not on PATH");
334    }
335
336    info!(target: "devflow", "using container engine: {}", cmd);
337    Ok(cmd.to_string())
338}

340/// Checks if an engine is not only installed but also has a responsive daemon.
341fn is_engine_healthy(name: &str) -> bool {
342    if !command_exists(name) {
343        return false;
344    }
345    // 'info' usually requires a working daemon link
346    Command::new(name)
347        .arg("info")
348        .stdout(std::process::Stdio::null())
349        .stderr(std::process::Stdio::null())
350        .status()
351        .map(|s| s.success())
352        .unwrap_or(false)
353}

355fn resolve_cache_root(cfg: &DevflowConfig, root: &str) -> PathBuf {
356    let path = PathBuf::from(root);
357    if path.is_absolute() {
358        return path;
359    }
360
361    let source_dir = cfg.source_dir.as_deref().unwrap_or_else(|| Path::new("."));
362    let abs_source = if source_dir.is_absolute() {
363        source_dir.to_path_buf()
364    } else {
365        std::env::current_dir().unwrap_or_default().join(source_dir)
366    };
367
368    abs_source.join(root)
369}

371fn parse_mount(mount: &str) -> Option<(&str, &str)> {
372    let parts: Vec<&str> = mount.split(':').collect();
373    if parts.len() == 2 {
374        Some((parts[0], parts[1]))
375    } else {
376        None
377    }
378}
