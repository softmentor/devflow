98fn main() -> Result<()> {
99    let format = fmt::format()
100        .with_target(false)
101        .with_level(true)
102        .with_timer(fmt::time::uptime())
103        .compact();
104
105    tracing_subscriber::registry()
106        .with(
107            fmt::layer()
108                .event_format(format)
109                .with_writer(std::io::stderr),
110        )
111        .with(EnvFilter::from_default_env().add_directive(tracing::Level::INFO.into()))
112        .init();
113
114    let cli = Cli::parse();
115    debug!("parsed cli arguments: {:?}", cli);
116
117    let command_name = match &cli.command {
118        Some(cmd) => cmd,
119        None => {
120            use clap::CommandFactory;
121            Cli::command().print_help()?;
122            println!(); // Add a newline after help
123            std::process::exit(0);
124        }
125    };
126
127    let command_text = match &cli.selector {
128        Some(selector) => format!("{}:{}", command_name, selector),
129        None => command_name.clone(),
130    };
131
132    let command = CommandRef::from_str(&command_text)
133        .map_err(|e| anyhow!("failed to parse command '{}': {e}", command_text))?;
134
135    if command.primary == PrimaryCommand::Init {
136        return init::run(&cli, command.selector.as_deref());
137    }
138
139    let cfg = DevflowConfig::load_from_file(&cli.config)
140        .with_context(|| format!("unable to load config '{}'", cli.config))?;
141    let mut registry = ExtensionRegistry::discover(&cfg)?;
142
143    // Phase 1 Wiring: Explicitly compile in the required trait implementations
144    registry.register(Box::new(devflow_ext_rust::RustExtension::new()));
145    registry.register(Box::new(devflow_ext_node::NodeExtension::new()));
146
147    // Phase 2 Wiring: Runtime discovery of Subprocess Extensions
148    discovery::discover_subprocess_extensions(&cfg, &mut registry)?;
149
150    registry.validate_target_support(&cfg)?;
151
152    execute(&cli, &cfg, &registry, &command)
153}

155/// Reports a GitHub status update.
156fn report_status(
157    context: &str,
158    state: &str,
159    description: &str,
160    target_url: Option<&str>,
161) -> Result<()> {
162    let token = match std::env::var("GITHUB_TOKEN") {
163        Ok(t) => t,
164        Err(_) => {
165            debug!("GITHUB_TOKEN not set, skipping status reporting");
166            return Ok(());
167        }
168    };
169
170    let repo = match std::env::var("GITHUB_REPOSITORY") {
171        Ok(r) => r,
172        Err(_) => {
173            debug!("GITHUB_REPOSITORY not set, skipping status reporting");
174            return Ok(());
175        }
176    };
177
178    let sha = std::env::var("GITHUB_HEAD_SHA")
179        .or_else(|_| std::env::var("GITHUB_SHA"))
180        .context("niether GITHUB_HEAD_SHA nor GITHUB_SHA is set")?;
181
182    let url = format!("https://api.github.com/repos/{}/statuses/{}", repo, sha);
183
184    let body = json!({
185        "state": state,
186        "context": context,
187        "description": description,
188        "target_url": target_url,
189    });
190
191    let resp = ureq::post(&url)
192        .set("Authorization", &format!("Bearer {}", token))
193        .set("Accept", "application/vnd.github+json")
194        .set("X-GitHub-Api-Version", "2022-11-28")
195        .send_json(body);
196
197    match resp {
198        Ok(_) => {
199            debug!(
200                "successfully reported status '{}' for context '{}'",
201                state, context
202            );
203            Ok(())
204        }
205        Err(e) => {
206            // We don't want to fail the whole command just because reporting failed,
207            // but we should log it.
208            tracing::warn!("failed to report status to GitHub: {}", e);
209            Ok(())
210        }
211    }
212}

223/// Executes a validated Devflow command.
224fn execute(
225    cli: &Cli,
226    cfg: &DevflowConfig,
227    registry: &ExtensionRegistry,
228    command: &CommandRef,
229) -> Result<()> {
230    if let Some(context) = &cli.report {
231        let target_url = get_gha_target_url();
232        report_status(
233            context,
234            "pending",
235            &format!("Running {}...", context),
236            target_url.as_deref(),
237        )?;
238
239        let result = execute_inner(cli, cfg, registry, command);
240
241        let (state, desc) = match &result {
242            Ok(_) => ("success", format!("{} passed", context)),
243            Err(_) => ("failure", format!("{} failed", context)),
244        };
245
246        report_status(context, state, &desc, target_url.as_deref())?;
247        result
248    } else {
249        execute_inner(cli, cfg, registry, command)
250    }
251}

253/// Internal execution logic.
254fn execute_inner(
255    cli: &Cli,
256    cfg: &DevflowConfig,
257    registry: &ExtensionRegistry,
258    command: &CommandRef,
259) -> Result<()> {
260    match command.primary {
261        PrimaryCommand::Check => {
262            let selector = command.selector.as_deref().unwrap_or("pr");
263            let resolved = devflow_policy::resolve_policy_commands(cfg, selector)?;
264            println!("check:{selector} (runtime={:?})", cfg.runtime.profile);
265            for cmd in resolved {
266                registry.ensure_can_run(&cmd)?;
267                println!(" - {}", cmd);
268                executor::run(cfg, registry, &cmd)?;
269            }
270            Ok(())
271        }
272        PrimaryCommand::Ci if command.selector.as_deref() == Some("generate") => {
273            let workflow = devflow_gh::render_workflow(cfg)?;
274            if cli.stdout {
275                println!("{workflow}");
276            } else {
277                write_ci_workflow(&cli.ci_output, &workflow)?;
278                println!("ci:generate wrote {}", cli.ci_output);
279            }
280            Ok(())
281        }
282        PrimaryCommand::Ci if command.selector.as_deref() == Some("check") => {
283            let expected = devflow_gh::render_workflow(cfg)?;
284            let actual = read_ci_workflow(&cli.ci_output)?;
285            devflow_gh::check_workflow(cfg, &actual)?;
286            if actual != expected {
287                return Err(anyhow!(
288                    "ci workflow drift detected in '{}': run 'dwf ci:generate' to resync",
289                    cli.ci_output
290                ));
291            }
292            println!("ci:check passed");
293            Ok(())
294        }
295        PrimaryCommand::Ci if command.selector.as_deref() == Some("plan") => {
296            let profiles = cfg
297                .targets
298                .profiles
299                .keys()
300                .cloned()
301                .collect::<Vec<_>>()
302                .join(", ");
303            println!("ci:plan profiles=[{}]", profiles);
304            Ok(())
305        }
306        PrimaryCommand::Prune => {
307            let selector = command.selector.as_deref().unwrap_or("cache");
308            match selector {
309                "cache" => {
310                    if cli.local || cli.all {
311                        let cache_dir = cfg
312                            .cache
313                            .as_ref()
314                            .and_then(|c| c.root.as_ref())
315                            .map(Path::new)
316                            .unwrap_or(Path::new(".cargo-cache"));
317                        let target_ci = Path::new("target/ci");
318
319                        let before_size = get_dir_size(cache_dir) + get_dir_size(target_ci);
320                        println!(
321                            "🧹 Pruning local caches (Current size: {} MB)...",
322                            before_size / 1024 / 1024
323                        );
324
325                        if cache_dir.exists() {
326                            fs::remove_dir_all(cache_dir).with_context(|| {
327                                format!("failed to remove cache dir '{}'", cache_dir.display())
328                            })?;
329                        }
330                        if target_ci.exists() {
331                            fs::remove_dir_all(target_ci)
332                                .with_context(|| "failed to remove target/ci")?;
333                        }
334
335                        let after_size = get_dir_size(cache_dir) + get_dir_size(target_ci);
336                        println!(
337                            "✨ Local cache pruned. (New size: {} MB, Reclaimed: {} MB)",
338                            after_size / 1024 / 1024,
339                            (before_size.saturating_sub(after_size)) / 1024 / 1024
340                        );
341                    }
342                    if (cli.gh || cli.all) && cli.force {
343                        let before_size = get_gh_cache_size().unwrap_or(0);
344                        println!(
345                            "🔥 Force-pruning ALL GitHub Actions caches (Current: {} MB)...",
346                            before_size / 1024 / 1024
347                        );
348                        run_gh_prune_cache(true)?;
349                        let after_size = get_gh_cache_size().unwrap_or(0);
350                        println!(
351                            "✨ All GH caches purged. (New size: {} MB)",
352                            after_size / 1024 / 1024
353                        );
354                    } else if cli.gh || cli.all {
355                        let before_size = get_gh_cache_size().unwrap_or(0);
356                        println!(
357                            "🧹 Pruning GitHub Actions caches (Current: {} MB)...",
358                            before_size / 1024 / 1024
359                        );
360                        run_gh_prune_cache(false)?;
361                        let after_size = get_gh_cache_size().unwrap_or(0);
362                        println!(
363                            "✨ GH caches pruned. (New size: {} MB, Reclaimed: {} MB)",
364                            after_size / 1024 / 1024,
365                            (before_size.saturating_sub(after_size)) / 1024 / 1024
366                        );
367                    }
368                }
369                "runs" => {
370                    if cli.gh || cli.all {
371                        let before_count = get_gh_run_count().unwrap_or(0);
372                        println!(
373                            "🧹 Pruning GitHub Actions workflow runs (Current: {} runs)...",
374                            before_count
375                        );
376                        run_gh_prune_runs()?;
377                        let after_count = get_gh_run_count().unwrap_or(0);
378                        println!(
379                            "✨ GH runs pruned. (New count: {}, Deleted: {})",
380                            after_count,
381                            before_count.saturating_sub(after_count)
382                        );
383                    }
384                }
385                _ => return Err(anyhow!("unknown prune selector '{}'", selector)),
386            }
387            Ok(())
388        }
389        _ => {
390            registry.ensure_can_run(command)?;
391            executor::run(cfg, registry, command)
392        }
393    }
394}

396fn run_gh_prune_cache(force: bool) -> Result<()> {
397    if force {
398        // Scorched Earth: Delete everything
399        let _ = std::process::Command::new("sh")
400            .arg("-c")
401            .arg("gh cache list --limit 100 --json id --jq '.[].id' | xargs -I {} gh cache delete {}")
402            .status()?;
403        return Ok(());
404    }
405
406    // 1. Stale PR cleanup (>24h)
407    let _ = std::process::Command::new("sh")
408        .arg("-c")
409        .arg("gh cache list --limit 100 --json id,ref,lastAccessedAt | jq -r '.[] | select(.ref | startswith(\"refs/pull/\")) | select((.lastAccessedAt | sub(\"\\\\.[0-9]+Z$\"; \"Z\") | fromdateiso8601) < (now - 86400)) | .id' | xargs -I {} gh cache delete {}")
410        .status()?;
411
412    // 2. Capacity-based pruning (>8GB)
413    let output = std::process::Command::new("sh")
414        .arg("-c")
415        .arg("gh cache list --limit 100 --json sizeInBytes --jq '[.[].sizeInBytes] | add // 0'")
416        .output()?;
417    let size_str = String::from_utf8_lossy(&output.stdout).trim().to_string();
418    let total_size: u64 = size_str.parse().unwrap_or(0);
419    let threshold: u64 = 8 * 1024 * 1024 * 1024; // 8GB
420
421    if total_size > threshold {
422        println!(
423            "⚠️ Cache limit reached ({} MB). Pruning refs...",
424            total_size / 1024 / 1024
425        );
426        let _ = std::process::Command::new("sh")
427            .arg("-c")
428            .arg("gh cache list --limit 100 --json ref --jq '.[].ref' | sort | uniq | xargs -I {ref} sh -c 'gh cache list --ref {ref} --json id,key | jq -r \".[] | select(.key | contains(\\\"cargo-\\\")) | .id\" | tail -n +2 | xargs -I {} gh cache delete {}'")
429            .status()?;
430    }
431    Ok(())
432}

434fn run_gh_prune_runs() -> Result<()> {
435    // 1. Failed/Canceled
436    let _ = std::process::Command::new("sh")
437        .arg("-c")
438        .arg("gh run list --status failure --limit 1000 --json databaseId --jq '.[].databaseId' | xargs -I {} gh run delete {}")
439        .status()?;
440    let _ = std::process::Command::new("sh")
441        .arg("-c")
442        .arg("gh run list --status cancelled --limit 1000 --json databaseId --jq '.[].databaseId' | xargs -I {} gh run delete {}")
443        .status()?;
444
445    // 2. Keep latest 100
446    let _ = std::process::Command::new("sh")
447        .arg("-c")
448        .arg("gh run list --limit 1000 --json databaseId --jq '.[].databaseId' | tail -n +101 | xargs -I {} gh run delete {}")
449        .status()?;
450    Ok(())
451}

453fn get_dir_size(path: &Path) -> u64 {
454    if !path.exists() {
455        return 0;
456    }
457    if path.is_file() {
458        return path.metadata().map(|m| m.len()).unwrap_or(0);
459    }
460    fs::read_dir(path)
461        .map(|entries| {
462            entries
463                .filter_map(|e| e.ok())
464                .map(|e| get_dir_size(&e.path()))
465                .sum()
466        })
467        .unwrap_or(0)
468}

492fn write_ci_workflow(path: &str, content: &str) -> Result<()> {
493    let output = Path::new(path);
494    if let Some(parent) = output.parent() {
495        fs::create_dir_all(parent)
496            .with_context(|| format!("failed to create directory '{}'", parent.display()))?;
497    }
498    fs::write(output, content).with_context(|| format!("failed to write '{}'", output.display()))
499}
