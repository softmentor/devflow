1//! Devflow extension for Node.js projects.
2//!
3//! Provides the [`NodeExtension`] which maps Devflow [`CommandRef`]s into practical
4//! `npm` commands, enabling JavaScript/TypeScript workflows to integrate transparently
5//! into the Devflow ecosystem.
6
7use anyhow::Result;
8use devflow_core::{CommandRef, ExecutionAction, Extension};
9use std::collections::HashSet;
10
11/// The Devflow extension for Node.js.
12///
13/// Discovers project capabilities and maps primary Devflow actions into
14/// localized `npm` invocations (e.g., `npm run build`, `npm ci`).
15#[derive(Debug, Default)]
16pub struct NodeExtension;
17
18impl NodeExtension {
19    /// Constructs a new [`NodeExtension`].
20    pub fn new() -> Self {
21        Self
22    }
23}
24
25impl Extension for NodeExtension {
26    fn name(&self) -> &str {
27        "node"
28    }
29
30    fn capabilities(&self) -> HashSet<String> {
31        [
32            "setup",
33            "fmt:check",
34            "fmt:fix",
35            "lint:static",
36            "build:debug",
37            "build:release",
38            "test:unit",
39            "test:integration",
40            "package:artifact",
41            "check",
42            "release",
43            "ci:generate",
44            "ci:check",
45        ]
46        .iter()
47        .map(|&s| s.to_string())
48        .collect()
49    }
50
51    fn build_action(&self, cmd: &CommandRef) -> Result<Option<ExecutionAction>> {
52        let primary = cmd.primary.as_str();
53        let selector = cmd.selector.as_deref().unwrap_or("");
54
55        let action = match (primary, selector) {
56            ("setup", "deps") => Some(action("npm", &["ci"])),
57            ("setup", "doctor") => Some(action("npm", &["--version"])),
58            ("fmt", "check") => Some(action("npm", &["run", "fmt:check"])),
59            ("fmt", "fix") => Some(action("npm", &["run", "fmt:fix"])),
60            ("lint", "static") => Some(action("npm", &["run", "lint"])),
61            ("build", "debug") => Some(action("npm", &["run", "build"])),
62            ("build", "release") => Some(action("npm", &["run", "build"])),
63            ("test", "unit") => Some(action("npm", &["run", "test:unit"])),
64            ("test", "integration") => Some(action("npm", &["run", "test:integration"])),
65            ("test", "smoke") => Some(action("npm", &["run", "test:smoke"])),
66            ("package", "artifact") => Some(action("npm", &["pack", "--dry-run"])),
67            _ => None,
68        };
69        Ok(action)
70    }
71
72    fn is_trusted(&self) -> bool {
73        true
74    }
75
76    fn cache_mounts(&self) -> Vec<String> {
77        vec!["node/npm:/root/.npm".to_string()]
78    }
79
80    fn env_vars(&self) -> std::collections::HashMap<String, String> {
81        let mut env = std::collections::HashMap::new();
82        env.insert("NPM_CONFIG_CACHE".to_string(), "/root/.npm".to_string());
83        env
84    }
85
86    fn fingerprint_inputs(&self) -> Vec<String> {
87        vec![
88            "package-lock.json".to_string(),
89            "yarn.lock".to_string(),
90            "pnpm-lock.yaml".to_string(),
91            "package.json".to_string(),
92        ]
93    }
94}
95
96/// Helper for constructing `ExecutionAction`s concisely.
97fn action(program: &str, args: &[&str]) -> ExecutionAction {
98    ExecutionAction {
99        program: program.to_string(),
100        args: args.iter().map(|s| s.to_string()).collect(),
101        env: std::collections::HashMap::new(),
102    }
103}

105#[cfg(test)]
106mod tests {
107    use super::*;
108    use devflow_core::PrimaryCommand;
109
110    fn cmd(primary: PrimaryCommand, selector: Option<&str>) -> CommandRef {
111        CommandRef {
112            primary,
113            selector: selector.map(|s| s.to_string()),
114        }
115    }

117    #[test]
118    fn smoke_test_extension_instantiation() {
119        let ext = NodeExtension::new();
120        assert_eq!(ext.name(), "node");
121    }
122
123    #[test]
124    fn unit_test_capabilities_exist() {
125        let ext = NodeExtension::new();
126        let caps = ext.capabilities();
127        assert!(caps.contains("build:debug"));
128        assert!(caps.contains("setup"));
129        assert!(caps.contains("lint:static"));
130    }
131
132    #[test]
133    fn unit_test_valid_build_actions() {
134        let ext = NodeExtension::new();
135
136        let tests = vec![
137            (cmd(PrimaryCommand::Setup, Some("deps")), "npm ci"),
138            (cmd(PrimaryCommand::Lint, Some("static")), "npm run lint"),
139            (cmd(PrimaryCommand::Test, Some("unit")), "npm run test:unit"),
140            (
141                cmd(PrimaryCommand::Package, Some("artifact")),
142                "npm pack --dry-run",
143            ),
144        ];
145
146        for (input_cmd, expected_shell) in tests {
147            let action = ext
148                .build_action(&input_cmd)
149                .expect("Expected valid action mapping")
150                .expect("Expected command to map to an action");
151            let actual_shell = format!("{} {}", action.program, action.args.join(" "));
152            assert_eq!(actual_shell, expected_shell);
153        }
154    }
155
156    #[test]
157    fn unit_test_invalid_build_actions_return_none() {
158        let ext = NodeExtension::new();
159
160        let invalid_cmds = vec![
161            cmd(PrimaryCommand::Fmt, Some("format")), // fmt:check and fmt:fix exist, not fmt:format
162            cmd(PrimaryCommand::Setup, Some("toolchain")),
163        ];
164
165        for input_cmd in invalid_cmds {
166            assert!(ext
167                .build_action(&input_cmd)
168                .expect("mapping should not error")
169                .is_none());
170        }
171    }
172
173    #[test]
174    fn is_trusted_returns_true() {
175        let ext = NodeExtension::new();
176        assert!(ext.is_trusted());
177    }
178
179    #[test]
180    fn cache_mounts_returns_expected_paths() {
181        let ext = NodeExtension::new();
182        let mounts = ext.cache_mounts();
183        assert_eq!(mounts.len(), 1);
184        assert_eq!(mounts[0], "node/npm:/root/.npm");
185    }
186
187    #[test]
188    fn env_vars_returns_expected_values() {
189        let ext = NodeExtension::new();
190        let envs = ext.env_vars();
191        assert_eq!(envs.get("NPM_CONFIG_CACHE").unwrap(), "/root/.npm");
192        assert_eq!(envs.len(), 1);
193    }
194
195    #[test]
196    fn fingerprint_inputs_returns_expected_files() {
197        let ext = NodeExtension::new();
198        let inputs = ext.fingerprint_inputs();
199        assert_eq!(inputs.len(), 4);
200        assert!(inputs.contains(&"package-lock.json".to_string()));
201        assert!(inputs.contains(&"yarn.lock".to_string()));
202        assert!(inputs.contains(&"pnpm-lock.yaml".to_string()));
203        assert!(inputs.contains(&"package.json".to_string()));
204    }
