1//! Devflow extension for Rust projects.
2//!
3//! Provides the [`RustExtension`] which maps Devflow [`CommandRef`]s into practical
4//! `cargo` commands, enabling Rust workflows to integrate transparently
5//! into the Devflow ecosystem.
6
7use anyhow::Result;
8use devflow_core::{CommandRef, ExecutionAction, Extension};
9use std::collections::HashSet;
10
11/// The Devflow extension for Rust.
12///
13/// Discovers project capabilities and maps primary Devflow actions into
14/// localized `cargo` invocations (e.g., `cargo build`, `cargo clippy`).
15#[derive(Debug, Default)]
16pub struct RustExtension;
17
18impl RustExtension {
19    /// Constructs a new [`RustExtension`].
20    pub fn new() -> Self {
21        Self
22    }
23}
24
25impl Extension for RustExtension {
26    fn name(&self) -> &str {
27        "rust"
28    }
29
30    fn capabilities(&self) -> HashSet<String> {
31        [
32            "setup",
33            "fmt:check",
34            "fmt:fix",
35            "lint:static",
36            "lint:security",
37            "build:debug",
38            "build:release",
39            "test:unit",
40            "test:integration",
41            "test:smoke",
42            "package:artifact",
43            "check",
44            "release",
45            "ci:generate",
46            "ci:check",
47        ]
48        .iter()
49        .map(|&s| s.to_string())
50        .collect()
51    }
52
53    fn build_action(&self, cmd: &CommandRef) -> Result<Option<ExecutionAction>> {
54        let primary = cmd.primary.as_str();
55        let selector = cmd.selector.as_deref().unwrap_or("");
56
57        let action = match (primary, selector) {
58            ("setup", "toolchain") => Some(action("rustup", &["show"])),
59            ("setup", "deps") => Some(action("cargo", &["fetch"])),
60            ("setup", "doctor") => Some(action("cargo", &["--version"])),
61            ("fmt", "check") => Some(action("cargo", &["fmt", "--all", "--", "--check"])),
62            ("fmt", "fix") => Some(action("cargo", &["fmt", "--all"])),
63            ("lint", "static") => Some(action(
64                "cargo",
65                &[
66                    "clippy",
67                    "--all-targets",
68                    "--all-features",
69                    "--",
70                    "-D",
71                    "warnings",
72                ],
73            )),
74            ("lint", "security") => Some(action(
75                "trivy",
76                &[
77                    "image",
78                    "devflow-ci:latest",
79                    "--severity",
80                    "CRITICAL,HIGH",
81                    "--exit-code",
82                    "1",
83                ],
84            )),
85            ("build", "debug") => Some(action("cargo", &["build"])),
86            ("build", "release") => Some(action("cargo", &["build", "--release"])),
87            ("test", "unit") => Some(action("cargo", &["nextest", "run", "--lib", "--bins"])),
88            ("test", "integration") => Some(action("cargo", &["test", "--tests"])),
89            ("test", "smoke") => Some(action("cargo", &["test", "smoke"])),
90            ("package", "artifact") => Some(action("cargo", &["build", "--release"])),
91            ("release", "candidate") => Some(action("cargo", &["build", "--release"])),
92            _ => None,
93        };
94        Ok(action)
95    }
96
97    fn is_trusted(&self) -> bool {
98        true
99    }
100
101    fn cache_mounts(&self) -> Vec<String> {
102        vec![
103            "rust/cargo:/workspace/.cargo-cache".to_string(),
104            "rust/target:/workspace/target/ci".to_string(),
105        ]
106    }
107
108    fn env_vars(&self) -> std::collections::HashMap<String, String> {
109        let mut env = std::collections::HashMap::new();
110        env.insert(
111            "CARGO_HOME".to_string(),
112            "/workspace/.cargo-cache".to_string(),
113        );
114        env.insert(
115            "CARGO_TARGET_DIR".to_string(),
116            "/workspace/target/ci".to_string(),
117        );
118        env.insert(
119            "SCCACHE_DIR".to_string(),
120            "/workspace/.cargo-cache/sccache".to_string(),
121        );
122        env.insert("RUSTC_WRAPPER".to_string(), "sccache".to_string());
123        env
124    }
125
126    fn fingerprint_inputs(&self) -> Vec<String> {
127        vec![
128            "Cargo.lock".to_string(),
129            "rust-toolchain.toml".to_string(),
130            "Cargo.toml".to_string(),
131        ]
132    }
133}
134
135/// Helper for constructing `ExecutionAction`s concisely.
136fn action(program: &str, args: &[&str]) -> ExecutionAction {
137    ExecutionAction {
138        program: program.to_string(),
139        args: args.iter().map(|s| s.to_string()).collect(),
140        env: std::collections::HashMap::new(),
141    }
142}

162    #[test]
163    fn unit_test_capabilities_exist() {
164        let ext = RustExtension::new();
165        let caps = ext.capabilities();
166        assert!(caps.contains("build:release"));
167        assert!(caps.contains("test:smoke"));
168        assert!(caps.contains("fmt:check"));
169        assert!(caps.contains("lint:security"));
170    }

172    #[test]
173    fn unit_test_valid_build_actions() {
174        let ext = RustExtension::new();
175
176        let tests = vec![
177            (
178                cmd(PrimaryCommand::Setup, Some("doctor")),
179                "cargo --version",
180            ),
181            (
182                cmd(PrimaryCommand::Build, Some("release")),
183                "cargo build --release",
184            ),
185            (
186                cmd(PrimaryCommand::Test, Some("integration")),
187                "cargo test --tests",
188            ),
189            (cmd(PrimaryCommand::Fmt, Some("fix")), "cargo fmt --all"),
190            (
191                cmd(PrimaryCommand::Test, Some("unit")),
192                "cargo nextest run --lib --bins",
193            ),
194            (
195                cmd(PrimaryCommand::Lint, Some("security")),
196                "trivy image devflow-ci:latest --severity CRITICAL,HIGH --exit-code 1",
197            ),
198        ];
199
200        for (input_cmd, expected_shell) in tests {
201            let action = ext
202                .build_action(&input_cmd)
203                .expect("Expected valid action mapping")
204                .expect("Expected command to map to an action");
205            let actual_shell = format!("{} {}", action.program, action.args.join(" "));
206            assert_eq!(actual_shell, expected_shell);
207        }
208    }
209
210    #[test]
211    fn unit_test_invalid_build_actions_return_none() {
212        let ext = RustExtension::new();
213
214        let invalid_cmds = vec![
215            cmd(PrimaryCommand::Build, Some("unknown-target")),
216            cmd(PrimaryCommand::Package, Some("docker")),
217        ];
218
219        for input_cmd in invalid_cmds {
220            assert!(ext

222                .expect("mapping should not error")
223                .is_none());
224        }
